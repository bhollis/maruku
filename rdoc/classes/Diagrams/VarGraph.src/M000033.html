                        def do_it(name2node, function)
                                dist2nodes = []
                                name2node.each do |var, node|
                                        dist = node.send function
                                        dist2nodes[dist] =[] if dist2nodes[dist] == nil
                                        dist2nodes[dist].push node 
                                end
                                dist2nodes
                        end
