        def iterate(pos1,pos2)
                inc_x = pos2[0]-pos1[0]
                inc_y = pos2[1]-pos1[1]
                len = [inc_x.abs,inc_y.abs].max + 1
                if inc_x.abs > inc_y.abs
                        len = inc_x.abs + 1
                else
                        len = inc_y.abs + 1
                end
                
                inc_x  /= inc_x.abs if inc_x != 0
                inc_y  /= inc_y.abs if inc_y != 0
                
#               raise "" if inc_x 
                for i in 0..(len-1)
                        x, y = pos1[0]+i*inc_x,pos1[1]+i*inc_y
                        yield x,y
                end
        end
