        def inspect2(valid)
                s = ""
                
                s += "+" + "-"*@width + "+\n"
                for y in 0..(@height-1)
                        s += "|"
                        for x in 0..(@width-1)
                                s += valid.get(x,y) ? get(x,y).chr : " "
                        end
                        s += "|"
                        s += "\n"
                end
                s += "+" + "-"*@width + "+\n"
                s
        end
