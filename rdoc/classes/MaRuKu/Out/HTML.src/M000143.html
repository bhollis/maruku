        def day_suffix(day)
                s = {
                        1 => 'st',
                        2 => 'nd',
                        3 => 'rd',
                        21 => 'st',
                        22 => 'nd',
                        23 => 'rd',
                        31 => 'st'
                }
                return s[day] || 'th';
        end
