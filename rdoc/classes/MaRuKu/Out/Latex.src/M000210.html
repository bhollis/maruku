        def latex_escape(source)
                s=""; 
                
                source.each_byte do |b| 
                        if b == ?\ 
                                s << '~'
                        elsif SAFE_CHARS.include? b
                                s << b
                        else
                                s += "\\char%d" % b 
                        end
                end
                s
        end
