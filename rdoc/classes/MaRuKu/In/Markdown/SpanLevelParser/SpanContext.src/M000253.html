                def describe
                        lines = @elements.map{|x| x.inspect}.join("\n")
                        s = "Elements read in span: \n" +
                        add_tabs(lines,1, ' -')+"\n"
                
                        if @cur_string.size > 0
                        s += "Current string: \n  #{@cur_string.inspect}\n" 
                        end
                        s
                end
