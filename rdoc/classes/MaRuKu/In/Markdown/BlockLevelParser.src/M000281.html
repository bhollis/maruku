        def read_metadata(src)
                hash = {}
                while src.cur_line 
                        case src.cur_line.md_type
                                when :empty;  src.shift_line
                                when :metadata; hash.merge! parse_metadata(src.shift_line)
                                else break
                        end
                end
                hash
        end
