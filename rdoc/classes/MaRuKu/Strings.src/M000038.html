        def strip_indent(s, n) 
                i = 0
                while i < s.size && n>0
                        c = s[i,1]
                        if c == ' '
                                n-=1;
                        elsif c == "\t"
                                n-=TabSize;
                        else
                                break
                        end
                        i+=1
                end
                s[i, s.size-1]
        end
