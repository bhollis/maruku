        def children_inspect(compact=true)
                s = @children.inspect_more(compact,', ')
                if @children.empty?
                        "[]"
                elsif s.size < 70
                        s
                else
                        "[\n"+
                        add_tabs(@children.inspect_more(compact,",\n",false))+
                        "\n]"
                end
        end
